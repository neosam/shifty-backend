    async fn book_slot_with_conflict_check(
        &self,
        booking: &service::booking::Booking,

    ) -> Result<BookingCreateResult, ServiceError> {

        // Permission HR ∨ self (Pattern S2 aus 03-PATTERNS.md)
        let (hr, sp) = tokio::join!(
            self.permission_service

                booking.sales_person_id,

                tx.clone().into(),

        );
        hr.or(sp)?;

        // Slot-Lookup
        let slot = self
            .slot_service

        // Date-Konversion (Pattern S3, D-Phase3-07)
        let booking_date: time::Date = time::Date::from_iso_week_date(
