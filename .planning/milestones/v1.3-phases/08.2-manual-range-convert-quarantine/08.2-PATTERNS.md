**Replacement-Pattern** (RESEARCH § 1 / Code-Examples Z. 412-459) — `match`-Branch über `manual_range`:
```rust
let (from_date, to_date) = match manual_range {
    Some(r) => {
        let fmt = time::macros::format_description!("[year]-[month]-[day]");
        let start = time::Date::parse(&r.start_date, fmt)
            .map_err(|_| ServiceError::ValidationError(/* InvalidValue: "manual_range.start_date" */))?;
        let end = time::Date::parse(&r.end_date, fmt)
            .map_err(|_| ServiceError::ValidationError(/* InvalidValue: "manual_range.end_date" */))?;
        let row_year = row.date_time.date().year();
        if start.year() != row_year || end.year() != row_year {
            return Err(ServiceError::ValidationError(/* InvalidValue: "year-mismatch" */));
        }
        let range = DateRange::new(start, end)
            .map_err(|_| ServiceError::DateOrderWrong(start, end))?;
        let dao_category = extra_hours_category_to_absence(&row.category);
        let conflicts = self.absence_dao
            .find_overlapping(row.sales_person_id, dao_category, range, None, tx.clone())
            .await?;
        if !conflicts.is_empty() {
            return Err(ServiceError::ValidationError(Arc::from([
                ValidationFailureItem::OverlappingPeriod(conflicts[0].logical_id),
            ])));
        }
        (start, end)
    }
    None => detect_weekly_lump_sum(&row, all_legacy.as_ref(), |d| {
        lookup_active_contract(work_details.as_ref(), d)
    }).ok_or_else(|| /* identical to existing */)?,
};
```
