```rust
//! Frontend state for the Cutover Wizard (Phase 8.1).
//!
//! Mirrors the wire DTOs (`CutoverProfileTO`, `CutoverGateDriftReportTO`,
//! `CutoverRunResultTO`) and adds wizard-specific UI state (current stage,
//! frontend-only skipped set per D-05, last-known summary for the success
//! page).

use std::sync::Arc;
use uuid::Uuid;
use rest_types::{
    CutoverGateDriftReportTO, CutoverProfileTO, CutoverRunResultTO,
};

#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub enum WizardStage {
    Profile,
    DryRun,
    Commit,
    Success,
}

impl Default for WizardStage {
    fn default() -> Self {
        WizardStage::Profile
    }
}

#[derive(Clone, Debug, PartialEq)]
pub struct RunSummary {
    pub total_clusters: u32,
    pub migrated_clusters: u32,
    pub quarantined_rows: u32,
    pub gate_drift_rows: u32,
    pub diff_report_path: Option<String>,
}

impl From<&CutoverRunResultTO> for RunSummary {
    fn from(r: &CutoverRunResultTO) -> Self {
        Self {
            total_clusters: r.total_clusters,
            migrated_clusters: r.migrated_clusters,
            quarantined_rows: r.quarantined_rows,
            gate_drift_rows: r.gate_drift_rows,
            diff_report_path: r.diff_report_path.as_deref().map(|s| s.to_string()),
        }
    }
}

#[derive(Clone, Debug, Default, PartialEq)]
pub struct CutoverWizardState {
    pub stage: WizardStage,
    pub profile: Option<CutoverProfileTO>,
    /// Latest `gate_drift_report` — fed inline from Convert/Bulk-Convert
    /// responses (D-08) or from a separate `gate-dry-run` call. `None` until

    pub last_run_summary: Option<RunSummary>,
    /// D-05: Skip is frontend-only — these IDs are filtered out of the visible
    /// drift list locally; they reappear after a page reload (no persistence).

    /// While a coroutine action is in flight (Convert / Edit / Delete /
    /// Bulk-Convert / RunDryRun): used by Plan 09 for the loading-spinner
    /// overlay (D-09).
    pub busy: bool,
}
```

```rust
//! Cutover wizard service-coroutine + global stores (Phase 8.1).
//!
//! Owns:
//! - `CUTOVER_STORE`: the wizard state mirror.
//! - `CUTOVER_DRIFT_REFRESH`: refresh-token bumped on each successful resolve
//!   action so the page re-renders the drift list (Plan 09 `use_effect`).
//! - `cutover_service`: the coroutine that processes `CutoverAction` events.
//!
//! Pattern mirror: `shifty-dioxus/src/service/absence.rs` (Plan 08-04 baseline).

use std::sync::Arc;
use dioxus::prelude::*;
use futures_util::StreamExt;
use rest_types::{AbsenceCategoryTO, ExtraHoursTO};
use tracing::info;
use uuid::Uuid;

use crate::api;
use crate::error::ShiftyError;
use crate::service::config::CONFIG;
use crate::service::error::{ErrorStore, ERROR_STORE};
use crate::state::cutover_state::{CutoverWizardState, RunSummary, WizardStage};

pub static CUTOVER_STORE: GlobalSignal<CutoverWizardState> =
    Signal::global(CutoverWizardState::default);

pub static CUTOVER_DRIFT_REFRESH: GlobalSignal<u64> = Signal::global(|| 0);

pub(crate) fn bump_cutover_refresh() {
    *CUTOVER_DRIFT_REFRESH.write() += 1;
}

#[derive(Debug)]
pub enum CutoverAction {
    LoadProfile,
    RunDryRun,
    Commit,
    ConvertSingle(Uuid),
    BulkConvert {
        sales_person_id: Uuid,
        category: AbsenceCategoryTO,
        year: u32,
    },
    UpdateExtraHours(ExtraHoursTO),
    DeleteExtraHours(Uuid),
    /// D-05: frontend-only — no API call.
    Skip(Uuid),
}

pub async fn cutover_service(mut rx: UnboundedReceiver<CutoverAction>) {
    while let Some(action) = rx.next().await {
        info!("CutoverAction: {:?}", &action);
        let config = CONFIG.read().clone();

        // Mark busy for spinner overlay (D-09). Skip is instant — no busy.
        if !matches!(action, CutoverAction::Skip(_)) {
            CUTOVER_STORE.write().busy = true;
        }

        let result = process_action(action, config).await;

        if let Err(err) = result {
            *ERROR_STORE.write() = ErrorStore { error: Some(err) };
        }
        CUTOVER_STORE.write().busy = false;
    }
}

async fn process_action(
    action: CutoverAction,
    config: crate::service::config::Config,
) -> Result<(), ShiftyError> {
    match action {
        CutoverAction::LoadProfile => {
            let profile = api::cutover_profile(config).await?;
            CUTOVER_STORE.write().profile = Some(profile);
        }
        CutoverAction::RunDryRun => {
            let result = api::cutover_gate_dry_run(config).await?;
            let mut store = CUTOVER_STORE.write();
            store.last_dry_run = result.gate_drift_report.clone();
            store.last_run_summary = Some(RunSummary::from(&result));
        }
        CutoverAction::Commit => {
            let result = api::cutover_commit(config).await?;
            let mut store = CUTOVER_STORE.write();
            store.last_run_summary = Some(RunSummary::from(&result));
            store.stage = WizardStage::Success;
        }
        CutoverAction::ConvertSingle(extra_hours_id) => {
            let resp = api::cutover_convert_quarantine_entry(config, extra_hours_id).await?;
            let mut store = CUTOVER_STORE.write();
            // D-08 inline pattern: write refreshed gate-drift-report directly.
            store.last_dry_run = resp.refreshed_drift_report;
            drop(store);
            bump_cutover_refresh();
        }
        CutoverAction::BulkConvert { sales_person_id, category, year } => {
            let resp = api::cutover_bulk_convert_quarantine_rows(
                config, sales_person_id, category, year, /*explicit_ids*/ None,
            ).await?;
            let mut store = CUTOVER_STORE.write();
            store.last_dry_run = resp.refreshed_drift_report;
            drop(store);
            bump_cutover_refresh();
        }
        CutoverAction::UpdateExtraHours(body) => {
            // Re-uses existing PUT /extra-hours/{id} (D-04).
            let _ = api::update_extra_hour(config.clone(), body).await?;
            // After edit, re-run gate-dry-run so the drift list reflects the
            // updated row (D-08 — Edit triggers Auto-Re-Run).
            let result = api::cutover_gate_dry_run(config).await?;
            CUTOVER_STORE.write().last_dry_run = result.gate_drift_report.clone();
            bump_cutover_refresh();
        }
        CutoverAction::DeleteExtraHours(id) => {
            api::delete_extra_hour(config.clone(), id).await?;
            // Same D-08 Auto-Re-Run.
            let result = api::cutover_gate_dry_run(config).await?;
            CUTOVER_STORE.write().last_dry_run = result.gate_drift_report.clone();
            bump_cutover_refresh();
        }
        CutoverAction::Skip(id) => {
            // D-05: frontend-only retain — no API call, no Auto-Re-Run.
            let mut store = CUTOVER_STORE.write();
            let mut next = store.skipped_extra_hours_ids.to_vec();
            next.push(id);
            store.skipped_extra_hours_ids = Arc::from(next.into_boxed_slice());
        }
    }
    Ok(())
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn bump_cutover_refresh_increments_token() {
        // GlobalSignal is shared across tests in the same process — capture
        // the current value first and assert delta of +1 after one bump.
        let before = *CUTOVER_DRIFT_REFRESH.read();
        bump_cutover_refresh();
        let after = *CUTOVER_DRIFT_REFRESH.read();
        assert_eq!(after, before + 1);
    }

    #[test]
    fn skip_action_appends_to_skipped_set() {
        // Reset the store (these tests share global state). Use a fresh Uuid.
        let id = Uuid::new_v4();
        // Reset the skipped set to empty for a clean assertion.
        CUTOVER_STORE.write().skipped_extra_hours_ids = Arc::from([] as [Uuid; 0]);
        // Call the action's handling logic directly. We don't spin up the
        // coroutine — that needs a Dioxus runtime. Instead exercise the
        // pure-state portion via `process_action` with a fake Config.
        //
        // process_action is async, so wrap in a minimal block_on. Use
        // tokio's current-thread runtime per existing test pattern.
        let runtime = tokio::runtime::Builder::new_current_thread()
            .enable_all().build().unwrap();
        runtime.block_on(async {
            // Minimal Config — Skip never reads it but the signature requires
            // one. Use an empty/default Config (verify the existing fn shape
            // — if Default isn't impl'd, construct manually with empty fields).
            let cfg = crate::service::config::Config::default();
            let _ = process_action(CutoverAction::Skip(id), cfg).await;
        });
        let store = CUTOVER_STORE.read();
        assert!(store.skipped_extra_hours_ids.iter().any(|x| *x == id));
    }
}
```
