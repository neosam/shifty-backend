```rust
    async fn bulk_convert_quarantine_rows(
        &self,
        sales_person_id: Uuid,
        category: crate::absence::AbsenceCategory,
        year: u32,
        explicit_ids: Option<Arc<[Uuid]>>,
        context: Authentication<Self::Context>,
        tx: Option<Self::Transaction>,
    ) -> Result<BulkConvertQuarantineRowsOutcome, ServiceError> {
        // 1. Privilege gate (D-23).
        self.permission_service
            .check_permission(CUTOVER_ADMIN_PRIVILEGE, context.clone())
            .await?;

        // 2. Open Tx.
        let tx = self.transaction_dao.use_transaction(tx).await?;

        // 3. Fetch all not-yet-migrated rows (heuristic needs siblings to detect
        //    cluster-of-N anyway).
        let all_legacy = self
            .cutover_dao
            .find_legacy_extra_hours_not_yet_migrated(tx.clone())
            .await?;

        // 4. Filter by triple. `category` is the service-tier enum — map the
        //    legacy row's `category` (which is an ExtraHoursCategoryEntity) via
        //    the existing `extra_hours_category_to_absence` helper for parity.
        let target_rows: Vec<_> = all_legacy
            .iter()
            .filter(|r| r.sales_person_id == sales_person_id)
            .filter(|r| extra_hours_category_to_absence(&r.category)
                        == category_to_dao(&category))
            .filter(|r| (r.date_time.date().year() as u32) == year)
            .filter(|r| match &explicit_ids {
                Some(ids) => ids.iter().any(|id| *id == r.id),
                None => true,
            })
            .cloned()
            .collect();

        if target_rows.is_empty() {
            return Err(ServiceError::EntityNotFoundGeneric(
                "No quarantined rows matched the (sales_person, category, year) triple".into(),
            ));
        }

        // 5. Resolve work_details once for the sales_person.
        let work_details = /* mirror Plan 02 Task 2 lookup; same call shape */;

        // 6. Per-row heuristic + accumulate. Strict-atomic (RESEARCH P-10) —
        //    any None aborts before any DAO writes.
        let now = self.clock.now();
        let synthetic_run_id = Uuid::new_v4();
        let mut planned: Vec<(LegacyExtraHoursRow, time::Date, time::Date)> = Vec::new();
        for row in target_rows.iter() {
            match detect_weekly_lump_sum(
                row,
                &all_legacy,
                |d| lookup_active_contract(&work_details, d),
            ) {
                Some((from, to)) => planned.push((row.clone(), from, to)),
                None => {
                    return Err(ServiceError::ValidationError(format!(
                        "Row {} does not match weekly-lump-sum heuristic; manual edit required",
                        row.id
                    )));
                }
            }
        }

        // 7. Apply all DAO writes inside the same Tx.
        let mut converted = Vec::with_capacity(planned.len());
        let mut deleted_ids: Vec<Uuid> = Vec::with_capacity(planned.len());
        for (row, from_date, to_date) in planned.iter() {
            let absence_period_id = Uuid::new_v4();
            let entity = AbsencePeriodEntity {
                id: absence_period_id,
                logical_id: absence_period_id,
                sales_person_id: row.sales_person_id,
                category: extra_hours_category_to_absence(&row.category),
                from_date: *from_date,
                to_date: *to_date,
                description: Arc::from(""),
                created: now,
                deleted: None,
                version: Uuid::new_v4(),
            };
            self.absence_dao
                .create(&entity, CUTOVER_MIGRATION_PROCESS, tx.clone())
                .await?;
            self.cutover_dao
                .upsert_migration_source(
                    &MigrationSourceRow {
                        extra_hours_id: row.id,
                        absence_period_id,
                        cutover_run_id: synthetic_run_id,  // RESEARCH Q3 — same id for whole batch
                        migrated_at: now,
                    },
                    tx.clone(),
                )
                .await?;
            converted.push(entity);
            deleted_ids.push(row.id);
        }

        // 8. Bulk soft-delete (one statement).
        self.extra_hours_service
            .soft_delete_bulk(
                Arc::from(deleted_ids.clone().into_boxed_slice()),
                CUTOVER_MIGRATION_PROCESS,
                Authentication::Full,
                Some(tx.clone()),
            )
            .await?;

        // 9. Inline refreshed drift-report (D-08).
        let (_clusters, _stats, bucket_map) = self
            .migrate_legacy_extra_hours_to_clusters(now, tx.clone())
            .await?;
        let refreshed = self
            .compute_gate_diagnostic(synthetic_run_id, now, &bucket_map, tx.clone())
            .await
            .ok();

        // 10. Atomic commit.
        self.transaction_dao.commit(tx).await?;

        Ok(BulkConvertQuarantineRowsOutcome {
            converted_absence_periods: converted,
            deleted_extra_hours_ids: deleted_ids,
            refreshed_drift_report: refreshed,
            errors: Vec::new(),  // strict-atomic — no per-row errors on success
        })
    }
```
