/// POST /admin/cutover/convert-quarantine-entry ‚Äî request body. Per D-01:
/// backend derives `(from, to)` via `detect_weekly_lump_sum` heuristic; the
/// frontend only supplies the row id.
#[derive(Clone, Debug, Serialize, Deserialize, ToSchema)]
pub struct CutoverConvertQuarantineEntryRequest {
    pub extra_hours_id: Uuid,
}

/// POST /admin/cutover/convert-quarantine-entry ‚Äî response body. Per D-08 the
/// `refreshed_drift_report` is included inline so the frontend skips a follow-up
/// `gate-dry-run` roundtrip after each Convert action. Old clients tolerate the
/// missing field via `#[serde(default)]`.
#[derive(Clone, Debug, Serialize, Deserialize, ToSchema)]
pub struct CutoverConvertQuarantineEntryResponse {
    pub absence_period: AbsencePeriodTO,
    pub deleted_extra_hours_id: Uuid,
    #[serde(default)]
    pub refreshed_drift_report: Option<CutoverGateDriftReportTO>,
}

/// POST /admin/cutover/bulk-convert-quarantine-rows ‚Äî request body. Per D-02:
/// either `(sales_person_id, category, year)` triple alone (= convert all
/// matching quarantine rows) OR `(triple) + extra_hours_ids` for explicit
/// subset selection. `extra_hours_ids = None` means "all of the triple".
#[derive(Clone, Debug, Serialize, Deserialize, ToSchema)]
pub struct CutoverBulkConvertQuarantineRowsRequest {
    pub sales_person_id: Uuid,
    pub category: AbsenceCategoryTO,
    pub year: u32,
    #[serde(default)]
    pub extra_hours_ids: Option<Vec<Uuid>>,
}

/// POST /admin/cutover/bulk-convert-quarantine-rows ‚Äî response body. Strict-
/// atomic Tx (D-02 + RESEARCH P-10): on heuristic mismatch the whole Tx rolls
/// back and the handler returns 422 ‚Äî `errors` is therefore expected to be
/// empty on a 200 response and present only when the implementation chose to
/// surface partial details for diagnostics. Field is kept in the success
/// response shape so future relaxations don't break the wire-contract.
#[derive(Clone, Debug, Serialize, Deserialize, ToSchema)]
pub struct CutoverBulkConvertQuarantineRowsResponse {
    pub converted_absence_periods: Vec<AbsencePeriodTO>,
    pub deleted_extra_hours_ids: Vec<Uuid>,
    #[serde(default)]
    pub refreshed_drift_report: Option<CutoverGateDriftReportTO>,
    #[serde(default)]
    pub errors: Vec<CutoverConvertErrorTO>,
}

/// Per-row failure detail used by `CutoverBulkConvertQuarantineRowsResponse.errors`.
#[derive(Clone, Debug, Serialize, Deserialize, ToSchema)]
pub struct CutoverConvertErrorTO {
    pub extra_hours_id: Uuid,
    pub reason: String,
}
