**Struktur** (Plan-Phase erweitert):
```rust
//! Cutover wizard coroutine service (Phase 8.1).
//!
//! Owns global `CUTOVER_STORE` (the wizard state) and `CUTOVER_DRIFT_REFRESH`
//! token (bumped on every successful Convert / Bulk-Convert / Edit / Delete
//! so the page can re-fetch dry-run inline).

use std::rc::Rc;
use dioxus::prelude::*;
use futures_util::StreamExt;
use uuid::Uuid;
use rest_types::{AbsenceCategoryTO, ExtraHoursTO};
use crate::{api, error::ShiftyError, state::cutover_state::CutoverWizardState};
use super::{config::CONFIG, error::{ErrorStore, ERROR_STORE}};

pub static CUTOVER_STORE: GlobalSignal<CutoverWizardState> =
    Signal::global(|| CutoverWizardState::default());

pub static CUTOVER_DRIFT_REFRESH: GlobalSignal<u64> = Signal::global(|| 0);

pub(crate) fn bump_cutover_refresh() { *CUTOVER_DRIFT_REFRESH.write() += 1; }

#[derive(Debug)]
pub enum CutoverAction {
    LoadProfile,
    RunDryRun,
    Commit,
    ConvertSingle(Uuid),                                    // extra_hours_id
    BulkConvert { sales_person_id: Uuid, category: AbsenceCategoryTO, year: u32 },
    UpdateExtraHours(ExtraHoursTO),                          // Inline-Edit
    DeleteExtraHours(Uuid),
    Skip(Uuid),                                              // Frontend-only D-05
}

pub async fn cutover_service(mut rx: UnboundedReceiver<CutoverAction>) {
    while let Some(action) = rx.next().await {
        let config = CONFIG.read().clone();
        match action {
            CutoverAction::LoadProfile => {
                match api::cutover_profile(config).await {
                    Ok(profile) => {
                        let mut store = CUTOVER_STORE.write();
                        store.profile = Some(profile);
                    }
                    Err(err) => *ERROR_STORE.write() = ErrorStore { error: Some(err) },
                }
            }
            CutoverAction::ConvertSingle(id) => {
                match api::cutover_convert_quarantine_entry(config, id).await {
                    Ok(resp) => {
                        // Inline drift-report aus der Response → kein separater Re-Run.
                        let mut store = CUTOVER_STORE.write();
                        store.last_dry_run = resp.refreshed_drift_report;
                        bump_cutover_refresh();
                    }
                    Err(err) => *ERROR_STORE.write() = ErrorStore { error: Some(err) },
                }
            }
            // ... weitere Branches analog zu absence_service.rs ...
            CutoverAction::Skip(id) => {
                // D-05: Frontend-only — append to skipped_set, no API call.
                let mut store = CUTOVER_STORE.write();
                let mut next = store.skipped_extra_hours_ids.to_vec();
                next.push(id);
                store.skipped_extra_hours_ids = next.into();
            }
        }
    }
}
```
