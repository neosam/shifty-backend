#[component]
pub fn CutoverAdminPage() -> Element {
    let auth = AUTH.read().clone();

    // Pitfall 4: AUTH-loading-Gate.
    if !auth.loading_done {
        return rsx! {
            TopBar {}
            div { class: "p-md text-ink-muted", "Loading..." }
        };
    }

    let is_cutover_admin = auth
        .auth_info
        .as_ref()
        .map(|a| a.has_privilege("cutover_admin"))
        .unwrap_or(false);
    let is_hr = auth
        .auth_info
        .as_ref()
        .map(|a| a.has_privilege("hr"))
        .unwrap_or(false);

    if !is_cutover_admin && !is_hr {
        return rsx! {
            TopBar {}
            div { class: "p-md text-bad", "Forbidden" }
        };
    }

    let i18n = I18N.read().clone();
    let cutover_handle = use_coroutine_handle::<CutoverAction>();
    let store = CUTOVER_STORE.read().clone();
    let already_done = FEATURE_FLAGS_STORE.read().absence_range_source_active();

    // Auto-load profile on first mount.
    let _profile_load = use_resource(move || async move {
        if CUTOVER_STORE.read().profile.is_none() {
            cutover_handle.send(CutoverAction::LoadProfile);
        }
    });

    // Subscribe to refresh-token bumps for re-renders. The store has already
    // been updated synchronously inside the coroutine; this effect is the
    // render trigger.
    let refresh_token = *CUTOVER_DRIFT_REFRESH.read();
    use_effect(move || { let _ = refresh_token; });

    rsx! {
        TopBar {}
        ErrorView {}
        div { class: "p-4 md:p-6 flex flex-col gap-3 max-w-5xl mx-auto",
            if already_done {
                IdempotenzBanner {}
            }
            header { class: "flex flex-col gap-1",
                h1 { class: "text-lg font-semibold text-ink", "{i18n.t(Key::CutoverPageTitle)}" }
                p  { class: "text-small text-ink-muted", "{i18n.t(Key::CutoverPageSubtitle)}" }
            }
            StageIndicator { active: store.stage, completed: completed_stages_for(&store) }
            div { class: "bg-surface border border-border rounded-lg p-6",
                {match store.stage {
                    WizardStage::Profile => rsx! { ProfileStage { store: store.clone() } },
                    WizardStage::DryRun  => rsx! { DryRunStage { store: store.clone(), is_cutover_admin: is_cutover_admin } },
                    WizardStage::Commit  => rsx! { CommitStage { store: store.clone(), is_cutover_admin: is_cutover_admin } },
                    WizardStage::Success => rsx! { SuccessPage { store: store.clone() } },
                }}
            }
            // Stage navigation footer (Back / Continue) — see Layout in UI-SPEC.
            StageNavFooter { store: store.clone() }
        }
    }
}

fn completed_stages_for(store: &CutoverWizardState) -> Vec<WizardStage> {
    let mut done = Vec::new();
    if matches!(store.stage, WizardStage::DryRun | WizardStage::Commit | WizardStage::Success)
        && store.profile.is_some()
    {
        done.push(WizardStage::Profile);
    }
    if matches!(store.stage, WizardStage::Commit | WizardStage::Success)
        && store
            .last_dry_run
            .as_ref()
            .map(|r| r.passed || r.total_drift_rows == 0)
            .unwrap_or(false)
    {
        done.push(WizardStage::DryRun);
    }
    if matches!(store.stage, WizardStage::Success) {
        done.push(WizardStage::Commit);
    }
    done
}

#[component]
fn StageIndicator(active: WizardStage, completed: Vec<WizardStage>) -> Element {
    let i18n = I18N.read().clone();
    let stages = [
        (WizardStage::Profile, Key::CutoverStage1Label),
        (WizardStage::DryRun,  Key::CutoverStage2Label),
        (WizardStage::Commit,  Key::CutoverStage3Label),
    ];
    rsx! {
        div { class: "flex items-center gap-0 w-full max-w-xl mx-auto my-6",
            for (idx, (stage, key)) in stages.iter().enumerate() {
                {
                    let is_active = active == *stage;
                    let is_complete = completed.contains(stage);
                    let circle_cls = match (is_active, is_complete) {
                        (true, _)  => "bg-accent text-accent-ink border-accent",
                        (_, true)  => "bg-good text-white border-good",
                        _          => "bg-surface text-ink-muted border-border",
                    };
                    let label_cls = match (is_active, is_complete) {
                        (true, _) => "text-accent",
                        (_, true) => "text-good",
                        _         => "text-ink-muted",
                    };
                    rsx! {
                        div { class: "flex flex-col items-center",
                            div { class: "w-8 h-8 rounded-full flex items-center justify-center text-small font-semibold border-2 {circle_cls}",
                                if is_complete { "✓" } else { "{idx + 1}" }
                            }
                            span { class: "text-small mt-1 {label_cls}",
                                "{i18n.t(*key)}"
                            }
                        }
                        if idx < stages.len() - 1 {
                            div { class: "flex-1 h-0.5 bg-border mx-2" }
                        }
                    }
                }
            }
        }
    }
}

#[component]
fn ProfileStage(store: CutoverWizardState) -> Element {
    let i18n = I18N.read().clone();
    let profile = store.profile.as_ref();
    let total_rows = profile.map(|p| p.total_legacy_rows).unwrap_or(0);
    let affected = profile
        .map(|p| {
            // Distinct sales_person_ids in buckets.
            let mut ids: Vec<Uuid> = p.buckets.iter().map(|b| b.sales_person_id).collect();
            ids.sort();
            ids.dedup();
            ids.len() as u32
        })
        .unwrap_or(0);
    let quarantine = profile.map(|p| p.total_quarantined_rows).unwrap_or(0);
    let carryover_diff = profile.map(|p| p.aggregate_carryover_diff).unwrap_or(0.0);
    rsx! {
        div { class: "flex flex-col gap-4",
            // 4 stat boxes — exact 4-col grid per UI-SPEC § Layout.
            div { class: "grid grid-cols-2 md:grid-cols-4 gap-3",
                StatBox { label: i18n.t(Key::CutoverStatTotalRows).into(), value: format!("{total_rows}").into() }
                StatBox { label: i18n.t(Key::CutoverStatPersons).into(),   value: format!("{affected}").into() }
                StatBox { label: i18n.t(Key::CutoverStatQuarantine).into(),value: format!("{quarantine}").into() }
                StatBox { label: i18n.t(Key::CutoverStatCarryoverDiff).into(), value: format!("{carryover_diff:.2}h").into() }
            }
            // Per-Person tile grid (PersonChip + per-person summary).
            if let Some(p) = profile {
                div { class: "grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-3",
                    for bucket in p.buckets.iter() {
                        // Inline the PersonChip pattern from existing pages
                        // (look up the existing component signature and use
                        // bucket.sales_person_name + bucket.row_count + bucket.legacy_sum_hours).
                        div { class: "bg-surface border border-border rounded-md p-3 flex flex-col gap-1",
                            span { class: "font-medium text-ink", "{bucket.sales_person_name}" }
                            span { class: "text-small text-ink-muted",
                                "{bucket.row_count} rows · {bucket.legacy_sum_hours:.2}h"
                            }
                        }
                    }
                }
            }
        }
    }
}

    let matched = input.read().as_str() == "CUTOVER";

                    StatBox { label: "Migrated clusters".into(), value: format!("{}", summary.migrated_clusters).into() }
                    StatBox { label: "Quarantined".into(),       value: format!("{}", summary.quarantined_rows).into() }
                }
            }
            // Type-to-confirm input.
            input {
                r#type: "text",
                class: "border border-border rounded-md p-2 text-base",
                placeholder: "CUTOVER",
                value: "{input}",
                oninput: move |ev| input.set(ev.value()),
            }
            // The actual confirm button — delegated to commit_form_inner so
            // tests can render it directly with controlled bools.
            { commit_form_inner(matched, is_cutover_admin) }
        }
    }
}

// Test-friendly helper — Task 2 Tests 6 + 7 call this directly.
fn commit_form_inner(matched: bool, is_admin: bool) -> Element {
    let i18n = I18N.read().clone();
    let cutover_handle = use_coroutine_handle::<CutoverAction>();
    let disabled = !matched || !is_admin;
    let tooltip = if !is_admin {
        Some(i18n.t(Key::CutoverPrivilegeStage3))
    } else if !matched {
        Some(i18n.t(Key::CutoverCommitTypeLabel))
    } else {
        None
    };
    rsx! {
        div { class: "flex flex-col gap-2",
            p { class: "text-small text-ink-muted", "{i18n.t(Key::CutoverCommitTypeLabel)}" }
            button {
                class: if disabled {
                    "px-4 py-2 rounded-md bg-bad/40 text-white cursor-not-allowed"
                } else {
                    "px-4 py-2 rounded-md bg-bad text-white hover:bg-bad/80"
                },
                disabled: disabled,
                onclick: move |_| {
                    if !disabled {
                        cutover_handle.send(CutoverAction::Commit);
                    }
                },
                "{i18n.t(Key::CutoverCommitBtn)}"
            }
            if let Some(t) = tooltip {
                span { class: "text-small text-ink-muted", "{t}" }
            }
        }
    }
}
