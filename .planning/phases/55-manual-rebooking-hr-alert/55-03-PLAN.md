//! VOL-ACCT-03 CI-Guard: two-part suite proving that a manual rebooking
//! roundtrip leaves the read aggregates `balance_hours`, `volunteer_hours`
//! and `committed_voluntary_target_in_range` invariant.
//!
//! Precondition: Plan 55-01 already applied the `source ==
//! ExtraHoursSource::Rebooking`-Filter in `service_impl/src/reporting.rs`

//! integration test below fails visibly.
//!
//! Design decision (WARNING #2, split-test):
//!   Part 1 (pure fn): proptest sweeps 128+ cases über
//!     Vec<ExtraHours> + apply the reporting-filter predicate; asserts sum
//!     equality with/without Rebooking-marker rows. Fast (no DB).
//!   Part 2 (integration): one deterministic case that boots in-memory sqlite,
//!     runs a real rebook_manual + reads back via ReportingService — verifies
//!     the pipeline end-to-end. Slow but single.

use proptest::prelude::*;

**Part 1 — Pure-fn Property-Test (schnell, 128 Cases):**

```
proptest! {
    #![proptest_config(ProptestConfig { cases: 128, .. ProptestConfig::default() })]

    #[test]
    fn reporting_filter_drops_rebooking_rows_regardless_of_pair_content(
        pair_hours in 0.01f32..50.0f32,
        baseline_manual in prop::collection::vec(-100.0f32..100.0f32, 0..8),
        direction_flag in prop::bool::ANY,
    ) {
        // Baue eine Vec<ExtraHours> mit `baseline_manual`-Werten
        // (source=Manual) + ein Rebooking-Pair (-pair, +pair) mit source=Rebooking.
        // Wende die Filter-Regel aus reporting.rs an
        //   (`.iter().filter(|eh| eh.source != ExtraHoursSource::Rebooking).collect()`).
        // Assert: sum(gefiltert) == sum(baseline_manual)  (Pair verschwindet).
        let filtered_sum: f32 = build_rows(&baseline_manual, pair_hours, direction_flag)
            .iter()
            .filter(|eh| eh.source != ExtraHoursSource::Rebooking)
            .map(|eh| eh.amount)
            .sum();
        let baseline_sum: f32 = baseline_manual.iter().sum();
        prop_assert!((filtered_sum - baseline_sum).abs() < 1e-3,
            "Filter ließ Rebooking-Row durch: {filtered_sum} vs {baseline_sum}");
    }

    #[test]
    fn proposed_hours_invariant(
        balance in -1000.0f32..1000.0f32,
        voluntary in 0.0f32..1000.0f32,
    ) {
        let ph = service::rebooking_reconciliation::proposed_rebooking_hours(balance, voluntary);
        prop_assert!(ph >= 0.0, "proposed_hours negativ");
        prop_assert!(ph <= balance.abs() + 1e-3, "proposed_hours > |balance|");
        prop_assert!(ph <= voluntary + 1e-3, "proposed_hours > voluntary_ist");
    }
}
```

**Part 2 — Deterministischer Integration-Test (1 Case, in-memory sqlite):**

```
#[tokio::test]
async fn manual_rebooking_roundtrip_leaves_aggregates_invariant() {
    let ctx = setup_environment_with_baseline(
        /* balance_seed */ -10.0,
        /* voluntary_seed */ 8.0,
    ).await;

    let (balance_before, voluntary_ist_before, voluntary_soll_before) =
        ctx.read_aggregates().await;

    ctx.rebook_manual(
        service::rebooking_reconciliation::RebookingDirection::VolunteerToExtra,
        3.0,
    ).await;

    let (balance_after, voluntary_ist_after, voluntary_soll_after) =
        ctx.read_aggregates().await;

    assert!((balance_before - balance_after).abs() < 1e-3);
    assert!((voluntary_ist_before - voluntary_ist_after).abs() < 1e-3);
    assert!((voluntary_soll_before - voluntary_soll_after).abs() < 1e-3);

    // Marker-Semantik: beide Pair-Rows tragen Rebooking; Baseline behält Manual.
    let rows = ctx.extra_hours_for_week().await;
    let rebooking_count = rows.iter()
        .filter(|eh| eh.source == ExtraHoursSource::Rebooking).count();
    assert_eq!(rebooking_count, 2, "erwarte genau 2 Marker-Rows");
    let manual_count = rows.iter()
        .filter(|eh| eh.source == ExtraHoursSource::Manual).count();
    assert!(manual_count >= 1, "Baseline-Row(s) sollten Manual bleiben");
}
```
