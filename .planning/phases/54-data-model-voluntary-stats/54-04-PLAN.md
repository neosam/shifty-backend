        #[instrument(skip(rest_state))]
        #[utoipa::path(
            get,
            path = "/{id}/voluntary-stats",
            tags = ["Report"],
            params(
                ("id" = Uuid, Path, description = "Sales person ID"),
                ("year" = u32, Query, description = "ISO-year for voluntary-stats aggregation")
            ),
            responses(
                (status = 200, description = "HR-only voluntary hours statistics; Non-HR gets all fields None", body = VoluntaryStatsTO, content_type = "application/json"),
                (status = 500, description = "Internal server error")
            )
        )]
        pub async fn get_voluntary_stats&lt;RestState: RestStateDef&gt;(
            rest_state: State&lt;RestState&gt;,
            query: Query&lt;VoluntaryStatsRequest&gt;,
            Path(sales_person_id): Path&lt;Uuid&gt;,
            Extension(context): Extension&lt;Context&gt;,
        ) -&gt; Response {
            error_handler(
                (async {
                    let stats = rest_state
                        .voluntary_stats_service()
                        .get_voluntary_stats(sales_person_id, query.year, context.into(), None)
                        .await?;
                    let to = VoluntaryStatsTO::from(&amp;stats);
                    Ok(Response::builder()
                        .status(200)
                        .header("Content-Type", "application/json")
                        .body(Body::new(serde_json::to_string(&amp;to).unwrap()))
                        .unwrap())
                })
                .await,
            )
        }
