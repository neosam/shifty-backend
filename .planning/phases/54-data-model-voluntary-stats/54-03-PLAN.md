    /// VOL-STAT-01 / VOL-ACCT-01-Ist (F1-Ist Zaehler): Summe der VolunteerWork-
    /// ExtraHours im ISO-Jahr `year`, gefiltert auf `source == Manual` (D-54-DM-02,
    /// Pitfall 1). Die ISO-Jahres-Zuordnung erfolgt via `ShiftyDate::from(...).as_shifty_week().year`
    /// — dieselbe Semantik wie im uebrigen reporting.rs.
    pub fn voluntary_ist_total_for_year(extra_hours: &amp;[ExtraHours], year: u32) -&gt; f32 {
        extra_hours
            .iter()
            .filter(|eh| matches!(eh.category, ExtraHoursCategory::VolunteerWork))
            .filter(|eh| eh.source == ExtraHoursSource::Manual)
            .filter(|eh| ShiftyDate::from(eh.date_time.date()).as_shifty_week().year == year)
            .map(|eh| eh.amount)
            .sum()
    }

    /// VOL-STAT-01-Nenner / VOL-ACCT-01-Ist_per_week Nenner (D-F1-01).
    /// Anzahl ISO-Wochen im Jahr, in denen mindestens eine gueltige EmployeeWorkDetails-Row
    /// den Kalender-Wochen-Slot abdeckt. `expected_hours == 0` zaehlt MIT (D-F1-01).
    /// Iteriert 1..=weeks_in_year(year).
    pub fn contract_weeks_count(working_hours: &amp;[EmployeeWorkDetails], year: u32) -&gt; u32 {
        let total = time::util::weeks_in_year(year as i32) as u8;
        (1..=total)
            .filter(|&amp;w| find_working_hours_for_calendar_week(working_hours, year, w).next().is_some())
            .count() as u32
    }

    /// D-F2-01 pro-rata: committed_voluntary fuer eine ISO-Woche, tagesweise verteilt.
    /// Iteriert Mo..So, waehlt fuer jeden Tag die aktive EmployeeWorkDetails-Row via
    /// `find_working_hours_for_calendar_week`, summiert (committed_voluntary / 7.0) je Tag.
    /// Guard: existiert die ISO-Woche fuer das Jahr nicht (z.B. 53 in einem 52-Wochen-Jahr),
    /// gibt 0.0 zurueck (Randfall).
    pub fn committed_voluntary_prorata_for_week(
        working_hours: &amp;[EmployeeWorkDetails],
        year: u32,
        week: u8,
    ) -&gt; f32 {
        if week &gt; time::util::weeks_in_year(year as i32) as u8 { return 0.0; }
        let Ok(monday) = time::Date::from_iso_week_date(year as i32, week, time::Weekday::Monday) else {
            return 0.0;
        };
        (0i64..7)
            .filter_map(|offset| {
                let day = monday + time::Duration::days(offset);
                let sw = ShiftyDate::from(day).as_shifty_week();
                find_working_hours_for_calendar_week(working_hours, sw.year, sw.week as u8)
                    .next()
                    .map(|wh| wh.committed_voluntary / 7.0)
            })
            .sum()
    }

    /// D-F2-01 Summe der pro-rata-Wochen ueber das ISO-Jahr.
    pub fn committed_voluntary_target_for_year(
        working_hours: &amp;[EmployeeWorkDetails],
        year: u32,
    ) -&gt; f32 {
        let total = time::util::weeks_in_year(year as i32) as u8;
        (1..=total)
            .map(|w| committed_voluntary_prorata_for_week(working_hours, year, w))
            .sum()
    }

      #[async_trait]
      impl&lt;Deps: VoluntaryStatsServiceDeps&gt; VoluntaryStatsService for VoluntaryStatsServiceImpl&lt;Deps&gt; {
          type Context = Deps::Context;

          async fn get_voluntary_stats(
              &amp;self,

              tx: Option&lt;Self::Transaction&gt;,
