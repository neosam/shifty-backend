          pub async fn load_voluntary_stats(
              config: Config,
              sales_person_id: Uuid,
              year: u32,
          ) -&gt; Result&lt;Option&lt;state::VoluntaryStats&gt;, ShiftyError&gt; {
              let url = format!("{}/report/{}/voluntary-stats?year={}", config.backend, sales_person_id, year);
              let resp = reqwest::Client::new().get(&amp;url).send().await?;
              if !resp.status().is_success() {
                  return Ok(None);
              }
              let to: VoluntaryStatsTO = resp.json().await?;
              // Non-HR: alle Felder None -&gt; Row wird nicht gerendert (siehe Task 3).
              // HR: mind. ist_per_contract_week Some.
              Ok(Some(state::VoluntaryStats::from(&amp;to)))
          }
