    ```rust
        /// Phase 5 helper (D-04, D-06): count active bookings in (slot_id, year, week)
        /// whose sales_person currently has `is_paid = true`. Soft-deleted bookings
        /// are filtered upstream by the booking_dao (WHERE deleted IS NULL); soft-

        /// (which goes through SalesPersonDao::all_paid → "WHERE deleted IS NULL AND is_paid = 1").
        /// Absence status is NOT considered (D-05: anyone booked counts).
        async fn count_paid_bookings_in_slot_week(
            &self,
            slot_id: uuid::Uuid,
            year: u32,
            week: u8,
            tx: Deps::Transaction,
        ) -> Result<u8, ServiceError> {
            let bookings = self
                .booking_service
                .get_for_week(week, year, Authentication::Full, Some(tx.clone()))
                .await?;
            let paid_persons = self
                .sales_person_service
                .get_all_paid(Authentication::Full, Some(tx.clone()))
                .await?;
            let paid_ids: std::collections::HashSet<uuid::Uuid> =
                paid_persons.iter().map(|sp| sp.id).collect();
            let count = bookings
                .iter()
                .filter(|b| b.slot_id == slot_id && b.deleted.is_none())
                .filter(|b| paid_ids.contains(&b.sales_person_id))
                .count();
            Ok(count.min(u8::MAX as usize) as u8)
        }
    }
    ```
